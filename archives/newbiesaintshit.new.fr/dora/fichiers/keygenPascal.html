program Keygen;
Uses crt;

var name : String;              // déclaration des variables
Var sum : integer;
Var var2 : integer;
Var i : integer;

begin

Writeln('Keygen - Dora L`exploratrice Keygenme -');
Writeln('========================================');
Writeln();
Write('Salut beau blond, comment tu s`appelle ? ');
Readln(name);

if length(name) < 6 then         // test sur la longueur du serial

   begin
   Writeln('name > 5 ! Sale gosse !');
   end

else if length(name) > 49 then

   begin
   Writeln('name < 50 ! Pfffff !');
   end

else                            // longueur correcte, on lance l'algo

   begin
   sum := 0;

   // somme d'un caractère sur deux du nom

   For i := 1 To length(name) Do sum := sum + Ord(name[i]);

   // effectue les opérations

   var2 := ((sum + $4D2B48) shl Ord(name[1])) XOR $6969;

   // affichage du serial

   Writeln('Le Serial est : D0r4-', var2);

   end;

Repeat Until KeyPressed;
